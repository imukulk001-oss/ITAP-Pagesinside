$(document).ready(function () {
  var heading = /**/ 'FT List';

  var table = $('#example0').DataTable({
    paging: false,
    dom: 'Bfrtip',
    select: true,
    buttons: [
      {
        extend: 'copy',
        text: 'Copy',
        title: heading
      },
      {
        extend: 'excel',
        text: 'Excel',
        title: heading,
        exportOptions: { columns: ':visible' }
      },
      {
        extend: 'print',
        text: 'Print Selected',
        title: heading,
        exportOptions: { columns: ':visible' }
      },
      {
        extend: 'print',
        text: 'Print All',
        title: heading,
        exportOptions: { columns: ':visible', modifier: { selected: null } }
      },
      {
        extend: 'pdfHtml5',
        text: 'PDF (Landscape)',
        title: '',
        orientation: 'landscape',
        pageSize: 'A4',
        exportOptions: { columns: ':visible' },
        customize: function (doc) {
          const currentUrl = window.location.href;
          const currentDate = new Date().toLocaleString();

          // === General Styles ===
          doc.defaultStyle.fontSize = 8;
          doc.styles.tableHeader.fontSize = 9;
          doc.styles.tableHeader.alignment = 'center';
          doc.styles.tableHeader.fillColor = '#0073AA';
          doc.styles.tableHeader.color = 'white';

          // === Reduce Margins to Use Full Page Width ===
          doc.pageMargins = [20, 50, 20, 40]; // [left, top, right, bottom]

          // === Add Title and Metadata Header ===
          doc.content.splice(0, 0, {
            columns: [
              { text: heading, fontSize: 14, bold: true, alignment: 'left' },
              { text: 'Downloaded from: ' + currentUrl + '\n' + currentDate, alignment: 'right', fontSize: 8 }
            ],
            margin: [0, 0, 0, 10]
          });

          // === Find Table Object ===
          let tableNode = null;
          for (let i = 0; i < doc.content.length; i++) {
            if (doc.content[i].table) {
              tableNode = doc.content[i];
              break;
            }
          }
          if (!tableNode || !tableNode.table || !tableNode.table.body) return;

          const body = tableNode.table.body;
          const colCount = body[0].length;

          // === Define Column Widths (Dynamic + Manual for Key Columns) ===
          const customWidths = [
            0,   // FT Id
            25,   // Id
            25,   // Actions
            0,   // CSIR Thematic Area
            35,   // Broad Area
            35,  // Project Title
            50,   // Project Category
            30,   // Lab
            20,   // PI
            30   // Status
            
          ];

          // Auto-fill remaining columns safely
          tableNode.table.widths = customWidths.concat(
            Array(Math.max(0, colCount - customWidths.length)).fill('*')
          );

          // === Center-align numeric data ===
          for (let i = 1; i < body.length; i++) {
            for (let j = 0; j < body[i].length; j++) {
              const txt = (body[i][j].text || '').toString().trim();
              if (txt && !isNaN(txt)) {
                body[i][j].alignment = 'center';
              }
            }
          }

          // === Enable Text Wrapping for Long Cells ===
          doc.styles.tableBodyEven = doc.styles.tableBodyEven || {};
          doc.styles.tableBodyOdd = doc.styles.tableBodyOdd || {};
          doc.styles.tableBodyEven.noWrap = false;
          doc.styles.tableBodyOdd.noWrap = false;
        }
      },
      'colvis',
      'pageLength'
    ]
  });

  // Append export buttons to top toolbar
  table.buttons().container().appendTo('#example_wrapper .col-md-6:eq(0)');
});
</script>

<script>
document.addEventListener("DOMContentLoaded", function () {
    const tbody = document.getElementById("fttdata");
    const rows = Array.from(tbody.querySelectorAll("tr"));

    let selectedFilters = {
        thematicArea: "",
        projectCategory: "",
        status: ""
    };

    function applyFilters() {
        let visibleCount = 0;

        rows.forEach(row => {
            const thematicArea = row.cells[4].innerText.trim();
            const projectCategory = row.cells[8].innerText.trim();
            const status = row.cells[11].innerText.trim();

            const matchThematicArea = !selectedFilters.thematicArea || thematicArea === selectedFilters.thematicArea;
            const matchProjectCategory = !selectedFilters.projectCategory || projectCategory === selectedFilters.projectCategory;
            const matchStatus = !selectedFilters.status || status === selectedFilters.status;

            const isVisible = matchThematicArea && matchProjectCategory && matchStatus;
            row.style.display = isVisible ? "" : "none";

            if (isVisible) visibleCount++;
        });

        // Show visible count
        document.getElementById("filterCount").innerText =
            `Showing ${visibleCount} of ${rows.length} records`;
    }

    function buildFilterButtons(columnIndex, filterDivId, keyName) {
        const filterDiv = document.getElementById(filterDivId);
        const counts = {};

        rows.forEach(row => {
            const value = row.cells[columnIndex].innerText.trim();
            counts[value] = (counts[value] || 0) + 1;
        });

        const allBtn = document.createElement("button");
        allBtn.className = "btn btn-outline-primary btn-sm active";
        allBtn.innerHTML = `All <span class="text-badge">${rows.length}</span>`;
        allBtn.dataset.value = "";
        allBtn.style = "font-weight: 600;font-size: 16px;margin-right:0.5rem;padding: 5px 25px;";
        filterDiv.appendChild(allBtn);

        Object.entries(counts).forEach(([value, count]) => {
            const btn = document.createElement("button");
            btn.className = "btn btn-outline-primary btn-sm";
            btn.innerHTML = `${value} <span class="text-badge">${count}</span>`;
            btn.dataset.value = value;
            btn.style = "font-weight: 600;font-size: 16px; margin-right:0.5rem; padding: 5px 25px;";
            filterDiv.appendChild(btn);
        });

        filterDiv.addEventListener("click", function (e) {
            const btn = e.target.closest("button");
            if (!btn) return;

            filterDiv.querySelectorAll("button").forEach(b => b.classList.remove("active", "btn-primary"));
            btn.classList.add("active", "btn-primary");

            selectedFilters[keyName] = btn.dataset.value;
            applyFilters();
        });
    }

    buildFilterButtons(4, "thematicAreaFilters", "thematicArea");
    buildFilterButtons(8, "projectCategoryFilters", "projectCategory");
    buildFilterButtons(11, "statusFilters", "status");

    applyFilters();
});
</script>

        allBtn.dataset.value = "";
        allBtn.innerHTML = `All <span class="text-badge">${rows.length}</span>`;
        allBtn.style = "font-weight:600;font-size:16px;margin-right:0.5rem;padding:5px 25px;";
        filterDiv.appendChild(allBtn);

        // Create contempt / non-contempt buttons
        Object.entries(counts).forEach(([key, count]) => {
            const btn = document.createElement("button");
            btn.className = "btn btn-outline-primary btn-sm";
            btn.dataset.value = key;
            btn.innerHTML = `${key} <span class="text-badge">${count}</span>`;
            btn.style = "font-weight:600;font-size:16px;margin-right:0.5rem;padding:5px 25px;";
            filterDiv.appendChild(btn);
        });

        // Button click handler
        filterDiv.addEventListener("click", function (e) {
            const btn = e.target.closest("button");
            if (!btn) return;

            filterDiv.querySelectorAll("button").forEach(b => b.classList.remove("active", "btn-primary"));
            btn.classList.add("active", "btn-primary");

            selectedFilter = btn.dataset.value;
            applyFilters();
        });
    }

    buildFilterButtons();
    applyFilters();
});
